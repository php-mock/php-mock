  9: /**
 10:  * Builds a sleep(), usleep(), time() and microtime() mock environment.
 11:  *
 12:  * In this environment sleep() and usleep() don't sleep for real. Instead
 13:  * they return immediatly and increase the amount of time in the mocks for
 14:  * time() and microtime().
 15:  *
 16:  * Example:
 17:  * <code>
 18:  * namespace foo;
 19:  *
 20:  * use malkusch\phpmock\SleepEnvironmentBuilder;
 21:  *
 22:  * $builder = new SleepEnvironmentBuilder();
 23:  * $builder->setNamespace(__NAMESPACE__)
 24:  *         ->setTimestamp(1417011228);
 25:  *
 26:  * $environment = $builder->build();
 27:  * $environment->enable();
 28:  *
 29:  * // This won't delay the test for 10 seconds, but increase time().
 30:  * sleep(10);
 31:  *
 32:  * assert(1417011228 + 10 == time());
 33:  * </code>
 34:  *
 35:  * @author Markus Malkusch <markus@malkusch.de>
 36:  * @link bitcoin:1335STSwu9hST4vcMRppEPgENMHD2r1REK Donations
 37:  * @license http://www.wtfpl.net/txt/copying/ WTFPL
 38:  */

 40: {
 41:

 43:      * @var string The namespace for the mock environment.
 44:      */

 48:      * @var mixed the timestamp.

 93:         $microtime = new FixedMicrotimeFunction($this->timestamp);
 94:         $builder->setName("microtime")
 95:                 ->setFunctionProvider($microtime);
 96:         $environment->addMock($builder->build());
 97:         
 98:         // time() mock
 99:         $builder->setName("time")
100:                 ->setFunction([$microtime, "getTime"]);
101:         $environment->addMock($builder->build());

103:         $incrementables = [$microtime];

105:         // sleep() mock
106:         $builder->setName("sleep")

108:         $environment->addMock($builder->build());
109:         
110:         // usleep() mock
111:         $builder->setName("usleep")
112:                 ->setFunctionProvider(new UsleepFunction($incrementables));
113:         $environment->addMock($builder->build());
